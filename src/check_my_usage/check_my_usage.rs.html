97fn format_number(n: u64) -> String {
98    let s = n.to_string();
99    let mut result = String::new();
100    for (i, c) in s.chars().rev().enumerate() {
101        if i > 0 && i % 3 == 0 {
102            result.insert(0, ',');
103        }
104        result.insert(0, c);
105    }
106    result
107}
