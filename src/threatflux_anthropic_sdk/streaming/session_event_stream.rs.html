1//! Streaming session events for Managed Agents (beta: managed-agents-2026-04-01).
2//!
3//! Unlike [`MessageStream`](crate::streaming::MessageStream), session events are
4//! delivered as complete JSON objects in each SSE `data:` frame — there is no
5//! multi-line content-block reassembly. This stream is therefore self-contained:
6//! it accumulates raw bytes, splits SSE frames on blank lines, and deserializes
7//! each frame's concatenated `data:` payload straight into a [`SessionEvent`].
8
9use crate::{
10    error::{AnthropicError, Result},
11    models::managed_agents::session_event::SessionEvent,
12};
13use futures::{Stream, StreamExt};
14use std::pin::Pin;
15use std::task::{Context, Poll};
16use tokio::sync::mpsc;
17
18/// Stream of [`SessionEvent`]s from a Managed Agents session.
19pub struct SessionEventStream {
20    receiver: mpsc::Receiver<Result<SessionEvent>>,
21    _handle: tokio::task::JoinHandle<()>,
22}
23
24impl SessionEventStream {
25    /// Create a new session-event stream from an HTTP response.
26    pub async fn new(response: reqwest::Response) -> Result<Self> {
27        let status = response.status();
28        if !status.is_success() {
29            let error_text = response.text().await.unwrap_or_default();
30            return Err(AnthropicError::api_error(status.as_u16(), error_text, None));
31        }
32
33        let (sender, receiver) = mpsc::channel(100);
34        let bytes_stream = response.bytes_stream();
35        let handle = tokio::spawn(pump_frames(bytes_stream, sender));
36
37        Ok(Self {
38            receiver,
39            _handle: handle,
40        })
41    }
42
43    /// Check if the stream is done.
44    pub fn is_done(&self) -> bool {
45        self.receiver.is_closed()
46    }
47}
48
49/// Read SSE byte chunks, accumulate frames, and forward decoded events.
50async fn pump_frames<S, B>(mut bytes_stream: S, sender: mpsc::Sender<Result<SessionEvent>>)
51where
52    S: Stream<Item = reqwest::Result<B>> + Unpin + Send + 'static,
53    B: AsRef<[u8]> + Send + 'static,
54{
55    let mut buffer = String::new();
56    while let Some(chunk_result) = bytes_stream.next().await {
57        match chunk_result {
58            Ok(chunk) => {
59                buffer.push_str(&String::from_utf8_lossy(chunk.as_ref()));
60                if !drain_frames(&mut buffer, &sender).await {
61                    return; // Receiver dropped or parse error — stop.
62                }
63            }
64            Err(e) => {
65                let error = AnthropicError::stream(format!("Stream chunk error: {}", e))
66                    .with_context("Session event stream processing");
67                let _ = sender.send(Err(error)).await;
68                return;
69            }
70        }
71    }
72    flush_trailing(&buffer, &sender).await;
73}
74
75/// Drain every complete frame from `buffer`, forwarding events. Returns `false`
76/// to stop the pump (the receiver was dropped, or a parse error was forwarded).
77async fn drain_frames(buffer: &mut String, sender: &mpsc::Sender<Result<SessionEvent>>) -> bool {
78    while let Some(boundary) = find_frame_boundary(buffer) {
79        let frame: String = buffer.drain(..boundary.end).collect();
80        let frame = &frame[..boundary.frame_len];
81        match parse_frame(frame) {
82            Ok(Some(event)) => {
83                if sender.send(Ok(event)).await.is_err() {
84                    return false;
85                }
86            }
87            Ok(None) => {} // Comment-only or `[DONE]` frame — skip.
88            Err(e) => {
89                let _ = sender.send(Err(e)).await;
90                return false;
91            }
92        }
93    }
94    true
95}
96
97/// Flush a trailing frame that was not terminated by a blank line.
98async fn flush_trailing(buffer: &str, sender: &mpsc::Sender<Result<SessionEvent>>) {
99    let trailing = buffer.trim();
100    if trailing.is_empty() {
101        return;
102    }
103    match parse_frame(trailing) {
104        Ok(Some(event)) => {
105            let _ = sender.send(Ok(event)).await;
106        }
107        Ok(None) => {}
108        Err(e) => {
109            let _ = sender.send(Err(e)).await;
110        }
111    }
112}

114/// Location of the first complete SSE frame in `buffer`.
115struct FrameBoundary {
116    /// Byte length of the frame content (excluding the blank-line separator).
117    frame_len: usize,
118    /// Byte index just past the blank-line separator (the drain point).
119    end: usize,
120}
121
122/// Find the first blank-line frame separator (`\n\n` or `\r\n\r\n`).
123fn find_frame_boundary(buffer: &str) -> Option<FrameBoundary> {
124    if let Some(pos) = buffer.find("\r\n\r\n") {
125        return Some(FrameBoundary {
126            frame_len: pos,
127            end: pos + 4,
128        });
129    }
130    if let Some(pos) = buffer.find("\n\n") {
131        return Some(FrameBoundary {
132            frame_len: pos,
133            end: pos + 2,
134        });
135    }
136    None
137}

139/// Parse a single SSE frame into a [`SessionEvent`].
140///
141/// Returns `Ok(None)` for comment-only frames or a `[DONE]` sentinel.
142fn parse_frame(frame: &str) -> Result<Option<SessionEvent>> {
143    let data = collect_sse_data(frame);
144    let data = data.trim();
145    if data.is_empty() || data == "[DONE]" {
146        return Ok(None);
147    }
148
149    let event = serde_json::from_str::<SessionEvent>(data).map_err(|e| {
150        AnthropicError::stream(format!("Failed to parse session event: {}", e))
151            .with_context("Session event deserialization")
152    })?;
153    Ok(Some(event))
154}

156/// Concatenate the `data:` payload lines of an SSE frame (newline-joined).
157///
158/// `event:` / `id:` / `retry:` / comment lines are ignored — the JSON payload is
159/// internally tagged on `type`, so the `event:` name is redundant.
160fn collect_sse_data(frame: &str) -> String {
161    let mut data = String::new();
162    for line in frame.lines() {
163        if let Some(rest) = sse_data_payload(line) {
164            if !data.is_empty() {
165                data.push('\n');
166            }
167            data.push_str(rest);
168        }
169    }
170    data
171}
172
173/// Extract the payload of a `data:` SSE line (allowing an optional leading
174/// space), or `None` for blank / comment / non-`data:` lines.
175fn sse_data_payload(line: &str) -> Option<&str> {
176    let line = line.trim_end_matches('\r');
177    let rest = line.strip_prefix("data:")?;
178    Some(rest.strip_prefix(' ').unwrap_or(rest))
179}

181impl Stream for SessionEventStream {
182    type Item = Result<SessionEvent>;
183
184    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
185        self.receiver.poll_recv(cx)
186    }
187}

189impl futures::stream::FusedStream for SessionEventStream {
190    fn is_terminated(&self) -> bool {
191        self.receiver.is_closed()
192    }
193}

199    #[test]
200    fn parse_frame_agent_message() {
201        let frame = "event: agent.message\ndata: {\"type\":\"agent.message\",\"id\":\"e1\",\"processed_at\":\"2026-04-01T00:00:00Z\",\"content\":[]}";
202        let event = parse_frame(frame).unwrap().unwrap();
203        assert!(matches!(event, SessionEvent::AgentMessage { .. }));
204    }
