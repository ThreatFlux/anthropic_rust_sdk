8/// A persistent memory store that can be attached to sessions.

14    /// Unique store identifier.
15    pub id: String,
16    /// Human-friendly name.
17    pub name: String,
18    /// Optional description.
19    #[serde(skip_serializing_if = "Option::is_none")]
20    pub description: Option<String>,
21    /// ISO 8601 creation timestamp.
22    #[serde(skip_serializing_if = "Option::is_none")]
23    pub created_at: Option<DateTime<Utc>>,
24    /// Additional fields not yet modeled explicitly.

204        let json = r#"{"type":"memory_store","id":"mem_1","name":"notes"}"#;
205        let store: MemoryStore = serde_json::from_str(json).unwrap();
206        assert_eq!(store.id, "mem_1");
207        assert_eq!(store.name, "notes");
208        let value = serde_json::to_value(&store).unwrap();
209        assert_eq!(value["type"], "memory_store");
210    }
