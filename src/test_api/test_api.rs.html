4use std::env;

7#[tokio::main]
8async fn main() -> Result<(), Box<dyn std::error::Error>> {
9    println!("\nüöÄ Threatflux SDK - API Test\n");
10
11    // Try to load from environment
12    let api_key = env::var("ANTHROPIC_API_KEY").unwrap_or_else(|_| {
13        eprintln!("‚ùå ANTHROPIC_API_KEY not set!");
14        eprintln!("Usage: ANTHROPIC_API_KEY=sk-ant-... cargo run --bin test_api");
15        std::process::exit(1);

18    println!("‚úÖ Using API key: {}...", &api_key[..10.min(api_key.len())]);

21    let config = Config::new(api_key)?;
22    let client = Client::new(config);
23
24    // Simple test
25    println!("\nTesting Messages API...");
26    let request = MessageBuilder::new()
27        .model("claude-3-5-haiku-20241022")
28        .max_tokens(30)
29        .user("Reply with: Hello, SDK is working!")
30        .build();
31
32    match client.messages().create(request, None).await {
33        Ok(response) => {
34            println!("‚úÖ Success!");
35            println!("Response: {}", response.text());
36            println!("Model: {}", response.model);
37            println!(
38                "Tokens: {} in, {} out",
39                response.usage.input_tokens, response.usage.output_tokens
40            );
41        }
42        Err(e) => {
43            eprintln!("‚ùå Error: {}", e);
44            eprintln!("Details: {:?}", e);
45        }
46    }
