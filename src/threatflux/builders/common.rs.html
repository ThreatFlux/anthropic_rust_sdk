5/// Common validation utilities for builders
6pub struct ValidationUtils;
7
8impl ValidationUtils {
9    /// Validate max_tokens parameter
10    pub fn validate_max_tokens(max_tokens: u32, context: &str) -> Result<(), AnthropicError> {
11        if max_tokens == 0 {
12            return Err(AnthropicError::invalid_input(format!(
13                "{} max_tokens must be greater than 0",
14                context
15            )));
16        }
17        Ok(())
18    }
19
20    /// Validate temperature parameter
21    pub fn validate_temperature(temperature: f32) -> Result<(), AnthropicError> {
22        if !(0.0..=1.0).contains(&temperature) {
23            return Err(AnthropicError::invalid_input(
24                "temperature must be between 0.0 and 1.0",
25            ));
26        }
27        Ok(())
28    }
29
30    /// Validate top_p parameter
31    pub fn validate_top_p(top_p: f32) -> Result<(), AnthropicError> {
32        if !(0.0..=1.0).contains(&top_p) {
33            return Err(AnthropicError::invalid_input(
34                "top_p must be between 0.0 and 1.0",
35            ));
36        }
37        Ok(())
38    }
39
40    /// Validate that messages are not empty
41    pub fn validate_messages_not_empty(
42        messages_len: usize,
43        context: &str,
44    ) -> Result<(), AnthropicError> {
45        if messages_len == 0 {
46            return Err(AnthropicError::invalid_input(format!(
47                "{} must contain at least one message",
48                context
49            )));
50        }
51        Ok(())
52    }
53
54    /// Validate Claude 4 specific constraints
55    pub fn validate_claude_4_constraints(
56        model: &str,
57        temperature: Option<f32>,
58        top_p: Option<f32>,
59    ) -> Result<(), AnthropicError> {
60        if model.starts_with("claude-opus-4-1") {
61            // Opus 4.1 cannot use both temperature and top_p simultaneously
62            if temperature.is_some() && top_p.is_some() {
63                return Err(AnthropicError::invalid_input(
64                    "Claude Opus 4.1 cannot use both temperature and top_p simultaneously",
65                ));
66            }
67        }
68        Ok(())
69    }
70
71    /// Validate thinking configuration
72    pub fn validate_thinking_config(
73        model: &str,
74        budget_tokens: Option<u32>,
75    ) -> Result<(), AnthropicError> {
76        if let Some(budget) = budget_tokens {
77            let max_allowed = crate::config::models::max_thinking_tokens(model).unwrap_or(0);
78            if max_allowed > 0 && budget > max_allowed {
79                return Err(AnthropicError::invalid_input(format!(
80                    "Thinking budget {} exceeds maximum {} for model {}",
81                    budget, max_allowed, model
82                )));
83            }
84        }
85        Ok(())
86    }
87}

103/// Preset configurations for consistent parameter combinations
104#[derive(Debug, Clone, Copy)]
105pub struct PresetConfig {
106    pub temperature: f32,
107    pub top_p: f32,
108    pub max_tokens: u32,
109}
110
111impl PresetConfig {
112    /// Creative writing preset
113    pub const CREATIVE: Self = Self {
114        temperature: 0.9,
115        top_p: 0.95,
116        max_tokens: 2000,
117    };
118
119    /// Analytical tasks preset
120    pub const ANALYTICAL: Self = Self {
121        temperature: 0.3,
122        top_p: 0.85,
123        max_tokens: 1500,
124    };
125
126    /// Code generation preset
127    pub const CODE_GENERATION: Self = Self {
128        temperature: 0.1,
129        top_p: 0.9,
130        max_tokens: 2000,
131    };
132
133    /// Conversational AI preset
134    pub const CONVERSATIONAL: Self = Self {
135        temperature: 0.7,
136        top_p: 0.9,
137        max_tokens: 1000,
138    };
139
140    /// Apply this preset to a builder with settable parameters
141    pub fn apply_to_builder<B: ParameterBuilder>(self, builder: B) -> B {
142        builder
143            .temperature(self.temperature)
144            .top_p(self.top_p)
145            .max_tokens(self.max_tokens)
146    }
147}
148
149/// Trait for builders that support parameter configuration
150pub trait ParameterBuilder: Sized {
151    fn temperature(self, temperature: f32) -> Self;
152    fn top_p(self, top_p: f32) -> Self;
153    fn max_tokens(self, max_tokens: u32) -> Self;
154
155    /// Apply a preset configuration
156    fn with_preset(self, preset: PresetConfig) -> Self {
157        preset.apply_to_builder(self)
158    }
159
160    /// Creative writing preset
161    fn creative(self) -> Self {
162        self.with_preset(PresetConfig::CREATIVE)
163    }
164
165    /// Analytical tasks preset
166    fn analytical(self) -> Self {
167        self.with_preset(PresetConfig::ANALYTICAL)
168    }
169
170    /// Code generation preset
171    fn code_generation(self) -> Self {
172        self.with_preset(PresetConfig::CODE_GENERATION)
173    }
174
175    /// Conversational AI preset
176    fn conversational(self) -> Self {
177        self.with_preset(PresetConfig::CONVERSATIONAL)
178    }
179}
