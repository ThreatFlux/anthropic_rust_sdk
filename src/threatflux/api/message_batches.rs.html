25    /// Create a message batch
26    ///
27    /// # Example
28    /// ```rust,no_run
29    /// use threatflux::{Client, Config, models::batch::MessageBatchCreateRequest};
30    ///
31    /// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
32    /// let client = Client::from_env()?;

37    /// println!("Created batch: {}", batch.id);
38    /// # Ok(())

41    pub async fn create(
42        &self,
43        request: MessageBatchCreateRequest,
44        options: Option<RequestOptions>,
45    ) -> Result<MessageBatch> {
46        let body = serde_json::to_value(request)?;
47        self.client
48            .request(HttpMethod::Post, "/messages/batches", Some(body), options)
49            .await
50    }

154    /// Wait for a batch to complete processing
155    pub async fn wait_for_completion(
156        &self,
157        batch_id: &str,
158        poll_interval: std::time::Duration,
159        max_wait: std::time::Duration,
160    ) -> Result<MessageBatch> {
161        let start_time = std::time::Instant::now();
162
163        loop {
164            let batch = self.retrieve(batch_id, None).await?;
165
166            match batch.processing_status {
167                MessageBatchStatus::Completed
168                | MessageBatchStatus::Failed
169                | MessageBatchStatus::Cancelled => {
170                    return Ok(batch);
171                }
172                _ => {
173                    if start_time.elapsed() >= max_wait {
174                        return Err(crate::error::AnthropicError::invalid_input(format!(
175                            "Batch {} did not complete within timeout",
176                            batch_id
177                        )));
178                    }
179
180                    tokio::time::sleep(poll_interval).await;
181                }
182            }
183        }
184    }

186    /// List batches by status
187    pub async fn list_by_status(
188        &self,
189        status: MessageBatchStatus,
190        options: Option<RequestOptions>,
191    ) -> Result<Vec<MessageBatch>> {
192        // This would typically involve API filtering, but for now we'll filter client-side
193        let pagination = create_default_pagination(None);
194        let response = self.list(Some(pagination), options).await?;
195
196        Ok(response
197            .data
198            .into_iter()
199            .filter(|batch| batch.processing_status == status)
200            .collect())
201    }
