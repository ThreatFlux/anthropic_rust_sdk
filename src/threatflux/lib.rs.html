1//! # Threatflux - Anthropic API Rust SDK
2//!
3//! A comprehensive Rust SDK for the Anthropic API, providing async support,
4//! streaming capabilities, and full coverage of the Anthropic API endpoints.
5//!
6//! ## Features
7//!
8//! - **Full API Coverage**: Complete implementation of all Anthropic API endpoints
9//! - **Streaming Support**: Real-time message streaming with Server-Sent Events
10//! - **Async/Await**: Built on tokio for high-performance async operations
11//! - **Rate Limiting**: Built-in rate limiting with configurable limits
12//! - **Retry Logic**: Exponential backoff retry with configurable policies
13//! - **Type Safety**: Strongly typed models for all API requests and responses
14//! - **File Uploads**: Support for uploading and managing files
15//! - **Batch Processing**: Process multiple messages in batches
16//! - **Admin API**: Full admin functionality for organizations and workspaces
17//! - **Vision Support**: Image processing capabilities with base64 encoding
18//! - **Tool Calling**: Function calling support with structured responses
19//!
20//! ## Quick Start
21//!
22//! ### Basic Message
23//! ```rust,no_run
24//! use threatflux::{Client, Config, models::MessageRequest};
25//!
26//! #[tokio::main]
27//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
28//!     let client = Client::from_env()?;
29//!     
30//!     let request = MessageRequest::new()
31//!         .model("claude-3-5-sonnet-20241022")
32//!         .max_tokens(1000)
33//!         .add_user_message("Hello, Claude!");
34//!     
35//!     let response = client.messages().create(request, None).await?;
36//!     println!("Response: {}", response.text());
37//!     
38//!     Ok(())
39//! }
40//! ```
41//!
42//! ### Streaming Messages
43//! ```rust,no_run
44//! use threatflux::{Client, models::MessageRequest};
45//! use futures::StreamExt;
46//!
47//! #[tokio::main]
48//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
49//!     let client = Client::from_env()?;
50//!     
51//!     let request = MessageRequest::new()
52//!         .model("claude-3-5-sonnet-20241022")
53//!         .max_tokens(1000)
54//!         .add_user_message("Tell me a story");
55//!     
56//!     let mut stream = client.messages().create_stream(request, None).await?;
57//!     
58//!     while let Some(event) = stream.next().await {
59//!         match event? {
60//!             threatflux::models::StreamEvent::ContentBlockDelta { delta, .. } => {
61//!                 if let Some(text) = delta.text {
62//!                     print!("{}", text);
63//!                 }
64//!             }
65//!             threatflux::models::StreamEvent::MessageStop => break,
66//!             _ => {}
67//!         }
68//!     }
69//!     
70//!     Ok(())
71//! }
72//! ```
73//!
74//! ### Configuration
75//! ```rust,no_run
76//! use threatflux::{Client, Config};
77//! use std::time::Duration;
78//!
79//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
80//! let config = Config::from_env()?
81//!     .with_timeout(Duration::from_secs(30))
82//!     .with_max_retries(3)
83//!     .with_rate_limit_rps(50);
84//!     
85//! let client = Client::try_new(config)?;
86//! # Ok(())
87//! # }
88//! ```
