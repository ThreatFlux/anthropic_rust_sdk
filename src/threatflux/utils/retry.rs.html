14/// A lightweight exponential backoff state machine used by the retry client.
15#[derive(Debug, Clone)]
16pub struct ExponentialBackoff {
17    /// Initial delay between retries.
18    pub initial_interval: Duration,
19    /// Maximum delay between retries.
20    pub max_interval: Duration,
21    /// Delay growth factor applied after each attempt.
22    pub multiplier: f64,
23    /// Maximum total time the backoff sequence may consume.
24    pub max_elapsed_time: Option<Duration>,
25    /// Delay that will be returned on the next retry.
26    pub current_interval: Duration,
27    /// Total time already spent in retry delays.
28    pub elapsed_time: Duration,
29}
30
31impl Default for ExponentialBackoff {
32    fn default() -> Self {
33        Self {
34            initial_interval: Duration::from_millis(500),
35            max_interval: Duration::from_secs(60),
36            multiplier: 1.5,
37            max_elapsed_time: Some(Duration::from_secs(900)),
38            current_interval: Duration::ZERO,
39            elapsed_time: Duration::ZERO,
40        }
41    }
42}
43
44impl ExponentialBackoff {
45    /// Return the next delay in the backoff sequence.
46    pub fn next_backoff(&mut self) -> Option<Duration> {
47        let base_interval = self.initial_interval.min(self.max_interval);
48        let delay = if self.current_interval == Duration::ZERO {
49            base_interval
50        } else {
51            self.current_interval.min(self.max_interval)
52        };
53
54        if let Some(limit) = self.max_elapsed_time {
55            if self.elapsed_time >= limit || self.elapsed_time.saturating_add(delay) > limit {
56                return None;
57            }
58        }
59
60        self.elapsed_time = self.elapsed_time.saturating_add(delay);
61
62        let multiplier = if self.multiplier.is_finite() && self.multiplier > 0.0 {
63            self.multiplier
64        } else {
65            1.0
66        };
67
68        let next_delay = Duration::from_secs_f64(
69            (delay.as_secs_f64() * multiplier).min(self.max_interval.as_secs_f64()),
70        );
71        self.current_interval = next_delay.max(base_interval);
72
73        Some(delay)
74    }
75}

77/// Client wrapper that adds retry logic to HTTP requests
78#[derive(Clone)]
79pub struct RetryClient {
80    http_client: HttpClient,
81    config: Arc<Config>,
82    stats: Arc<std::sync::Mutex<RetryStats>>,
83}
84
85impl RetryClient {
86    /// Create a new retry client
87    pub fn new(config: Arc<Config>) -> Self {
88        let http_client = HttpClient::new(config.clone());
89
90        Self {
91            http_client,
92            config,
93            stats: Arc::new(std::sync::Mutex::new(RetryStats::default())),
94        }
95    }
96
97    /// Make an HTTP request with retry logic
98    pub async fn request<T>(
99        &self,
100        method: HttpMethod,
101        url: &Url,
102        body: Option<serde_json::Value>,
103        headers: HeaderMap,
104        timeout: Duration,
105    ) -> Result<T>
106    where
107        T: DeserializeOwned,
108    {
109        let _start_time = std::time::Instant::now();
110        let mut backoff = self.create_backoff();
111
112        // Update total requests stat
113        {
114            let mut stats = self.stats.lock().unwrap();
115            stats.total_requests += 1;
116        }
117
118        // Track attempt statistics
119
120        for attempt in 0..=self.config.max_retries {
121            match self
122                .http_client
123                .request(method, url, body.clone(), headers.clone(), timeout)
124                .await
125            {
126                Ok(result) => {
127                    if attempt == 0 {
128                        let mut stats = self.stats.lock().unwrap();
129                        stats.successful_first_try += 1;
130                    } else {
131                        let mut stats = self.stats.lock().unwrap();
132                        stats.retried_requests += 1;
133                        stats.total_retry_attempts += attempt as u64;
134                    }
135                    return Ok(result);
136                }
137                Err(error) => {
138                    // Store error for potential return later
139
140                    // Don't retry on final attempt
141                    if attempt == self.config.max_retries {
142                        let mut stats = self.stats.lock().unwrap();
143                        stats.failed_requests += 1;
144                        return Err(error);
145                    }
146
147                    // Check if we should retry this error
148                    if !self.should_retry(&error) {
149                        let mut stats = self.stats.lock().unwrap();
150                        stats.failed_requests += 1;
151                        return Err(error);
152                    }
153
154                    // Calculate delay
155                    let delay = self.calculate_delay(&error, &mut backoff);
156
157                    tracing::debug!(
158                        "Request failed (attempt {}/{}), retrying in {:?}: {}",
159                        attempt + 1,
160                        self.config.max_retries + 1,
161                        delay,
162                        error
163                    );
164
165                    // Update retry delay stats
166                    {
167                        let mut stats = self.stats.lock().unwrap();
168                        stats.total_retry_delay += delay;
169                    }
170
171                    tokio::time::sleep(delay).await;
172                }
173            }
174        }
175
176        // This should never be reached due to loop structure
177        Err(AnthropicError::Unknown(anyhow::anyhow!(
178            "All retry attempts failed"
179        )))
180    }
181
182    /// Create exponential backoff configuration
183    fn create_backoff(&self) -> ExponentialBackoff {
184        ExponentialBackoff {
185            initial_interval: Duration::from_millis(1000),
186            max_interval: Duration::from_secs(60),
187            multiplier: 2.0,
188            max_elapsed_time: Some(Duration::from_secs(300)), // 5 minutes total
189            ..Default::default()
190        }
191    }
192
193    /// Determine if an error should trigger a retry
194    fn should_retry(&self, error: &AnthropicError) -> bool {
195        match error {
196            AnthropicError::Http(reqwest_error) => {
197                // Retry on network errors (connection failed, timeout, etc.)
198                reqwest_error.is_timeout()
199                    || reqwest_error.is_connect()
200                    || reqwest_error.is_request()
201            }
202            AnthropicError::Api { status, .. } => {
203                // Retry on specific HTTP status codes
204                HttpClient::should_retry(*status)
205            }
206            AnthropicError::RateLimit(_) => true,
207            AnthropicError::Timeout(_) => true,
208            _ => false,
209        }
210    }
211
212    /// Calculate delay before next retry attempt
213    fn calculate_delay(
214        &self,
215        error: &AnthropicError,
216        backoff: &mut ExponentialBackoff,
217    ) -> Duration {
218        match error {
219            AnthropicError::RateLimit(_) => {
220                // For rate limit errors, use a longer delay
221                Duration::from_secs(60)
222            }
223            AnthropicError::Api { status: 429, .. } => {
224                // 429 Too Many Requests - use exponential backoff but start with longer delay
225                backoff.next_backoff().unwrap_or(Duration::from_secs(30))
226            }
227            AnthropicError::Api { status, .. } if HttpClient::is_server_error(*status) => {
228                // Server errors - use exponential backoff
229                backoff.next_backoff().unwrap_or(Duration::from_secs(30))
230            }
231            _ => {
232                // Default exponential backoff
233                backoff.next_backoff().unwrap_or(Duration::from_secs(1))
234            }
235        }
236    }
237
238    /// Create a smart backoff that considers rate limit headers
239    pub fn create_smart_backoff(&self, rate_limit_info: &RateLimitInfo) -> Duration {
240        // If we have explicit retry-after header, use it
241        if let Some(retry_after) = rate_limit_info.retry_after {
242            return retry_after;
243        }
244
245        // If we have rate limit info, calculate intelligent delay
246        if let Some(delay) = rate_limit_info.recommended_delay() {
247            return delay;
248        }
249
250        // Fallback to default exponential backoff
251        Duration::from_secs(1)
252    }
253
254    /// Get retry statistics
255    pub fn stats(&self) -> RetryStats {
256        self.stats.lock().unwrap().clone()
257    }
258
259    /// Reset retry statistics
260    pub fn reset_stats(&self) {
261        let mut stats = self.stats.lock().unwrap();
262        *stats = RetryStats::default();
263    }
264}

266/// Retry policy configuration
267#[derive(Debug, Clone)]
268pub struct RetryPolicy {
269    /// Maximum number of retry attempts
270    pub max_retries: u32,
271    /// Initial delay between retries
272    pub initial_delay: Duration,
273    /// Maximum delay between retries
274    pub max_delay: Duration,
275    /// Backoff multiplier
276    pub backoff_multiplier: f64,
277    /// Maximum total time to spend retrying
278    pub max_elapsed_time: Option<Duration>,
279    /// Jitter to add to delays (prevents thundering herd)
280    pub jitter: bool,
281}
282
283impl Default for RetryPolicy {
284    fn default() -> Self {
285        Self {
286            max_retries: 3,
287            initial_delay: Duration::from_millis(1000),
288            max_delay: Duration::from_secs(60),
289            backoff_multiplier: 2.0,
290            max_elapsed_time: Some(Duration::from_secs(300)),
291            jitter: true,
292        }
293    }
294}
295
296impl RetryPolicy {
297    /// Create a new retry policy
298    pub fn new() -> Self {
299        Self::default()
300    }
301
302    /// Set maximum number of retries
303    pub fn with_max_retries(mut self, max_retries: u32) -> Self {
304        self.max_retries = max_retries;
305        self
306    }
307
308    /// Set initial delay
309    pub fn with_initial_delay(mut self, delay: Duration) -> Self {
310        self.initial_delay = delay;
311        self
312    }
313
314    /// Set maximum delay
315    pub fn with_max_delay(mut self, delay: Duration) -> Self {
316        self.max_delay = delay;
317        self
318    }
319
320    /// Set backoff multiplier
321    pub fn with_backoff_multiplier(mut self, multiplier: f64) -> Self {
322        self.backoff_multiplier = multiplier;
323        self
324    }
325
326    /// Set maximum elapsed time
327    pub fn with_max_elapsed_time(mut self, duration: Duration) -> Self {
328        self.max_elapsed_time = Some(duration);
329        self
330    }
331
332    /// Enable/disable jitter
333    pub fn with_jitter(mut self, jitter: bool) -> Self {
334        self.jitter = jitter;
335        self
336    }
337
338    /// Create exponential backoff from this policy
339    pub fn create_backoff(&self) -> ExponentialBackoff {
340        ExponentialBackoff {
341            initial_interval: self.initial_delay,
342            max_interval: self.max_delay,
343            multiplier: self.backoff_multiplier,
344            max_elapsed_time: self.max_elapsed_time,
345            ..Default::default()
346        }
347    }
348}

350/// Retry statistics for monitoring
351#[derive(Debug, Clone, Default)]
352pub struct RetryStats {
353    /// Total number of requests attempted
354    pub total_requests: u64,
355    /// Number of requests that succeeded on first try
356    pub successful_first_try: u64,
357    /// Number of requests that required retries
358    pub retried_requests: u64,
359    /// Number of requests that failed after all retries
360    pub failed_requests: u64,
361    /// Total number of retry attempts made
362    pub total_retry_attempts: u64,
363    /// Total time spent waiting for retries
364    pub total_retry_delay: Duration,
365}
366
367impl RetryStats {
368    /// Get success rate (requests that eventually succeeded)
369    pub fn success_rate(&self) -> f64 {
370        if self.total_requests == 0 {
371            return 1.0;
372        }
373
374        let successful = self.total_requests - self.failed_requests;
375        successful as f64 / self.total_requests as f64
376    }
377
378    /// Get retry rate (requests that required retries)
379    pub fn retry_rate(&self) -> f64 {
380        if self.total_requests == 0 {
381            return 0.0;
382        }
383
384        self.retried_requests as f64 / self.total_requests as f64
385    }
386
387    /// Get average retries per request
388    pub fn average_retries(&self) -> f64 {
389        if self.total_requests == 0 {
390            return 0.0;
391        }
392
393        self.total_retry_attempts as f64 / self.total_requests as f64
394    }
